/* MEMBER */
CREATE TABLE TABLE (
   MEMBER_SEQ NUMBER NOT NULL, /* 회원 번호 */
   ID VARCHAR2(400) NOT NULL, /* 아이디 */
   PASSWORD VARCHAR2(400) NOT NULL, /* 비밀번호 */
   NAME VARCHAR2(400) NOT NULL, /* 이름 */
   AGE NUMBER NOT NULL, /* 나이 */
   EMAIL VARCHAR2(400) NOT NULL, /* 이메일 */
   PHONE NUMBER NOT NULL, /* 전화번호 */
   KIND VARCHAR2(400) NOT NULL, /* 회원구분 */
   ADDRESS VARCHAR2(400) NOT NULL /* 주소 */
);

COMMENT ON TABLE TABLE IS 'MEMBER';

COMMENT ON COLUMN TABLE.MEMBER_SEQ IS '회원 번호';

COMMENT ON COLUMN TABLE.ID IS '아이디';

COMMENT ON COLUMN TABLE.PASSWORD IS '비밀번호';

COMMENT ON COLUMN TABLE.NAME IS '이름';

COMMENT ON COLUMN TABLE.AGE IS '나이';

COMMENT ON COLUMN TABLE.EMAIL IS '이메일';

COMMENT ON COLUMN TABLE.PHONE IS '전화번호';

COMMENT ON COLUMN TABLE.KIND IS '회원구분';

COMMENT ON COLUMN TABLE.ADDRESS IS '주소';

CREATE UNIQUE INDEX PK_TABLE
   ON TABLE (
      MEMBER_SEQ ASC
   );

ALTER TABLE TABLE
   ADD
      CONSTRAINT PK_TABLE
      PRIMARY KEY (
         MEMBER_SEQ
      );
